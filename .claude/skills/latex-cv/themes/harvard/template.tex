% ── Section heading: hrule above, bold centered title ───────────────────────
\newcommand{\ressection}[1]{%
  \vspace{14pt}%
  \hrule height 0.4pt%
  \vspace{8pt}%
  {\centering\textbf{\large #1}\par}%
  \vspace{6pt}%
}

% ── Experience entry ─────────────────────────────────────────────────────────
% With location: \jobentry{Company}{Location}{Title}{Dates}
% Use when location adds signal (foreign company, remote, multi-office)
\newcommand{\jobentry}[4]{%
  \noindent\textbf{#1} \hfill \textcolor[gray]{0.4}{#2}\par\vspace{-2pt}%
  \noindent\textbf{#3} \hfill \textcolor[gray]{0.4}{#4}\par\vspace{1pt}%
}

% Without location: \jobentrynoloc{Company}{Title}{Dates}
% Default — all on one line: Company (bold) --- Title (normal weight), Dates right-aligned
% \par at end is critical — without it the next content continues inline
\newcommand{\jobentrynoloc}[3]{%
  \noindent\textbf{#1} --- #2 \hfill \textcolor[gray]{0.4}{#3}\par\vspace{1pt}%
}

% ── Sub-project within a company ─────────────────────────────────────────────
% Use when a company has multiple named projects; order sub-projects by JD relevance
% No trailing \vspace — topsep=3pt on the following itemize provides the gap
% With dates: \subproject{Project Name}{Dates}
\newcommand{\subproject}[2]{%
  \vspace{3pt}%
  \noindent\textit{#1} \hfill \textcolor[gray]{0.4}{#2}\par%
}
% Without dates: \subprojectnodates{Project Name}
\newcommand{\subprojectnodates}[1]{%
  \vspace{3pt}%
  \noindent\textit{#1}\par%
}

  parsep     = 0pt,

}
